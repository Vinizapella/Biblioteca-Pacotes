# 1. Inicia o git na sua pasta
git init

# 2. Renomeia a branch principal
git branch -M main

# 3. Adiciona todos os arquivos
git add .

# 4. Salva localmente
git commit -m "Primeiro commit"

# 5. Conecta no GitHub (TROQUE A URL)
git remote add origin https://github.com/usuario/repo.git

# 6. Envia para o GitHub
git push -u origin main
